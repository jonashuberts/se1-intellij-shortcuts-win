\section{1. Sonderzeichen (Syntax)}

\textbf{Achtung:} Syntax-Zeichen benötigen meist \fbox{\textbf{Alt Gr}}.

\vspace{0.1cm}

\begin{center}

\small
\begin{tabular}{p{0.22\columnwidth} p{0.33\columnwidth} p{0.33\columnwidth}}

\textbf{|} (Pipe) & \textbf{Alt Gr + <} & Logisches ODER \\
\textbf{\textbackslash} & \textbf{Alt Gr + ?} & Escape, Pfade \\
\textbf{@} & \textbf{Alt Gr + Q} & Annotationen \\
\bottomrule
\end{tabular}
\end{center}

\section{5. Troubleshooting \& Debugger}

\small

\textbf{Roter Code / Fehler?}
\begin{itemize}[leftmargin=*, noitemsep]
    \item Cursor auf Fehler $\to$ \fbox{\textbf{Alt + Enter}}
    \item Syntax prüfen: \texttt{;} oder \texttt{\}} vergessen?
\end{itemize}

\textbf{Parameter anzeigen}
\begin{itemize}[leftmargin=*, noitemsep]
    \item Cursor in \texttt{( )} $\to$ \textbf{Strg + P}
\end{itemize}

\textbf{Fenster maximiert?}
\begin{itemize}[leftmargin=*, noitemsep]
    \item Toggle Editor-Maximierung: \textbf{Shift + F12}
\end{itemize}

\textbf{Debugger}
\begin{itemize}[leftmargin=*, noitemsep]
    \item \textbf{F7} Step Into (In Methode rein)
    \item \textbf{F8} Step Over (Nächste Zeile)
    \item \textbf{F9} Resume (Weiter bis Breakpoint)
\end{itemize}
